## 1.

```js
var a = 10,
  b = 11,
  c = 12;
function test(a) {
  a = 1;
  var b = 2;
  c = 3;
}
test(10);
console.log(a, b, c);
//=> 10 11 3
```

## 3.

```js
var a = 9;
function fn() {
  a = 0;
  return function(b) {
    return b + a++;
  };
}
var f = fn();
console.log(f(5));
console.log(fn()(5));
console.log(f(5));
console.log(a);
//=> 5
//=> 5
//=> 6
//=> 2
```
